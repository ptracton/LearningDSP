Let's start to learn and try out Matlab
x = [-pi:.1:pi];
ysin = sin(x);
ycos = cos(x);
plot(x,ysin, x,ycos);
xlabel('X Axis');
ylabel('Y Axis');
title('Sine and Cosine from -\pi to \pi');
legend({'Sine','Cosine'});
